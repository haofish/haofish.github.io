using namespace std;

const int INF = 0x3f3f3f3f;
int m, n, g[15][110], ans[15][110], Min, dp[15][110], f;
bool vis[15][110];

int DP(int r, int c) {
    if (c == n) return 0;
    if (vis[r][c]) return dp[r][c];
    int &res = dp[r][c], M = INF, &i = ans[r][c + 1], row[] = {r - 1, r, r + 1};
    vis[r][c] = true, res = g[r][c];
    if (r == 0) row[0] = m - 1;
    if (r == m - 1) row[2] = 0;
    sort(row, row + 3);
    for (int j = 0; j < 3; j++)
        if (M > DP(row[j], c + 1)) M = DP(row[j], c + 1), i = row[j];
    res += M;
    return res;
}

using namespace std;

const int M = 5e3 + 10, INF = 0x3f3f3f3f;
int add[M][M], dp[M][M];

struct strings{
    int s, t;
    strings(int s = INF, int t = 0) : s(s), t(t) {}
};


void init(char *a, char *b) {
    int n = strlen(a + 1), m = strlen(b + 1);
    strings aa[27], bb[27];
    for (int i = 1; i <= n; i++) {
        strings &t = aa[a[i] - 'A'];
        t.s = min(t.s, i), t.t = i;
    }
    for (int i = 1; i <= m; i++) {
        strings &t = bb[b[i] - 'A'];
        t.s = min(t.s, i), t.t = i;
    }
    for (int i = 0; i <= n; i++) {
        for (int j = 0; j <= m; j++) {
            add[i][j] = 0;
            bool ok = true;
            if (i) {
                add[i][j] = add[i - 1][j];
                int x = a[i] - 'A';
                if (aa[x].s == i && bb[x].s > j) add[i][j]++;
                if (aa[x].t <= i && bb[x].t <= j) add[i][j]--;
                ok = false;
            }
            if (ok && j) {
                add[i][j] = add[i][j - 1];
                int x = b[j] - 'A';
                if (bb[x].s == j && aa[x].s > i) add[i][j]++;
                if (bb[x].t <= j && aa[x].t <= i) add[i][j]--;
            }
            // printf("%d ", add[i][j]);
        }
        // puts("");
    }
    // printf("%d", aa['R' - 'A'].t);
}


int DP(char *a, char *b) {
    int n = strlen(a + 1), m = strlen(b + 1);
    dp[0][0] = 0;
    for (int i = 0; i <= n; i++) {
        for (int j = 0; j <= m; j++) {
            if (i) {
                dp[i][j] = dp[i - 1][j] + add[i - 1][j];
            }
            if (j) {
                if (i) dp[i][j] = min(dp[i][j], dp[i][j - 1] + add[i][j - 1]);
                else dp[i][j] = dp[i][j - 1] + add[i][j - 1];
            }
        }
    }
    return dp[n][m];
}

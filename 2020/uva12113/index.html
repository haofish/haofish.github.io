#include <cstdio>
#include <algorithm>
#include <cstring>

using namespace std;

char g[10][20], p[10][20];
bool vis[10];

bool init() {
    for (int i = 0; i < 5; i++) {
        fgets(g[i], 15, stdin);
        if (g[i][0] == '0') return false;
    }
    memset(p, ' ', sizeof p);
    memset(vis, 0, sizeof vis);
    return true;
}

bool check() {
    for (int i = 0; i < 5; i++) {
        for (int j = 0; j < 9; j++) if (p[i][j] != g[i][j]) return false;
    }
    return true;
}

bool dfs(int num) {
    if (check()) return true;
    if (num == 6) return false;
    char tmp[10][20];
    memcpy(tmp, p, sizeof p);
    for (int i = 0; i < 9; i++) {
        if (vis[i]) continue;
        vis[i] = true;
        int r = i / 3, c = (i % 3) * 2;
        p[r + 1][c] = p[r + 2][c] = p[r + 1][c + 4] = p[r + 2][c + 4] = '|';
        p[r][c + 1] = p[r + 2][c + 1] = p[r][c + 3] = p[r + 2][c + 3] = '_';
        for (int j = 1; j < 4; j++) p[r + 1][c + j] = ' ';
        p[r + 2][c + 2] = ' ';
        if (dfs(num + 1)) return true; 
        vis[i] = false;
        memcpy(p, tmp, sizeof p);//å›æº¯
    }
    return false;
}

Parentheses Balance

#include <bits/stdc++.h>

using namespace std;

int t;

int main () {
#ifndef ONLINE_JUDGE
    freopen("test.in", "r", stdin);
    freopen("test.out", "w", stdout);
#endif
    scanf("%d", &t);
    getchar();
    while (t--) {
        char str[150];
        fgets(str, sizeof str, stdin);
        stack<char> st;
        for (int i = 0; str[i] != '\n'; i++) {
            if (str[i] == ')' && st.size() && st.top() == '(') st.pop();
            else if (str[i] == ']' && st.size() && st.top() == '[') st.pop();
            else st.push(str[i]);
        }
        if (st.empty()) puts("Yes");
        else puts("No");
    }
    return 0;
}

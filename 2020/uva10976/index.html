Fractions Again?!

#include <cstdio>
#include <vector>

using namespace std;
using ll = long long;

ll n;

int main() {
#ifndef ONLINE_JUDGE
    freopen("test.in", "r", stdin);
    freopen("test.out", "w", stdout);
#endif
    while (~scanf("%lld", &n)) {
        vector<ll> a, b;
        for (ll i = n + 1; i <= n + n; i++) {
            if (n * i % (i - n) == 0) {
                a.push_back(n * i / (i - n));
                b.push_back(i);
            }
        }
        printf("%zu\n", a.size());
        for (size_t i = 0; i < a.size(); i++) {
            printf("1/%lld = 1/%lld + 1/%lld\n", n, a[i], b[i]);
        }
    }
    return 0;
}

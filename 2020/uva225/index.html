#include <cstdio>
#include <algorithm>
#include <cstring>
#include <vector>

using namespace std;

const int M = 1e3;
int t, k, bn, num;
bool isb[M + M][M + M], vis[M + M][M + M];
const char dir[] = "ewns";
const int dx[] = {1, -1, 0, 0};
const int dy[] = {0, 0, 1, -1};
vector<char> ans[1000], tmp;

void init() {
    num = 0;
    memset(isb, 0, sizeof isb);
    memset(vis, 0, sizeof vis);
    for (auto &i : ans) i.clear();
}

void dfs(int a, int b, int n, int d) {
    if (n == k) {
        if (a != M || b != M) return;
        ans[num++] = tmp;
        return;
    }
    for (int i = d; i < 4 - 2 + d; i++) {
        int newx = a, newy = b;
        bool flag = true;
        for (int j = 0; j <= n; j++) {//èµ°n+1æ­¥
            newx += dx[i], newy += dy[i];
            if (isb[newx][newy]) { flag = false; break; }//é‡åˆ°éšœç¢å°±ä¸è¡Œ
        }
        if (!flag || vis[newx][newy]) continue;
        vis[newx][newy] = true;
        tmp.push_back(dir[i]);
        dfs(newx, newy, n + 1, 2 - d);
        tmp.pop_back();
        vis[newx][newy] = false;
    }
}

int main () {

            isb[x + M][y + M] = true;//éšœç¢
        }
        dfs(M, M, 0, 0);
        dfs(M, M, 0, 2);

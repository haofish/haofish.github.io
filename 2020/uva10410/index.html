int n, x;

    for (; ~scanf("%d", &n);) {
        vector<int> node[1010], dfs(n), bfs(n + 1);
        for (int i = 1; i <= n; i++) {
            scanf("%d", &x);
            bfs[x] = i;
        }
        for (auto &i : dfs) scanf("%d", &i);
        stack<int> st;
        int root = dfs[0];
        st.push(root);
        for (int i = 1; i < n; i++) {
            for (;;) {
                int fa = st.top();
                if (bfs[fa] + 1 < bfs[dfs[i]] || fa == root) {
                    node[fa].push_back(dfs[i]);
                    st.push(dfs[i]);
                    break;
                } else {
                    st.pop();
                }
            }
        }
        for (int i = 1; i <= n; i++) {
            printf("%d:", i);
            for (auto it : node[i]) {
                printf(" %d", it);
            }
            puts("");
        }
    }
    return 0;
}

二分

    1. 二分分法：[l, mid) 和 [mid, r],目标值符合右区间及其端点
int bsearch(int l, int r){
    while (l < r) {
        int mid = (long long)(l + r) >> 1;//相当于 l + r 后向下取整
        if (/* 探针落在左区间 */) l = mid + 1；
        else r = mid;
    }
    return l;
}
    1. 二分分法：[l, mid] 和 (mid, r], 目标值符合左区间及其端点
int bsearch(int l, int r){
    while (l < r) {
        int mid = (long long)(l + r + 1) >> 1; //相当于 l + r 后向上取整，否则陷入死循环，例如 此时 l = 1, r = 2, 那么探针就会一直落到左区间l上（l + r >> 1 =  1），下一步 l = mid = l,就会死循环
        if (/* 探针落在左区间 */) l = mid;
        else r = mid - 1;
    }
    return l;
}

快速幂

int qpow(int a, int b, int mod){
    int ans = 1 % mod;
    while(b != 0) {
        if (b & 1) ans = (long long)(ans * a) % mod;
        b >>= 1;
        a = (long long)(a * a) % mod;
    }
    return ans;
}

最大公因数，最小公倍数

//最大公因数，公约数
int gcd(int a, int b){
    int m;
    do{
        m = a % b;
        a = b;
        b = m;
    }while(m != 0);
    return a;
}

//最小公倍数
int lcm(int a, int b) {
    return a / gcd(a, b) * b;
}

int bsearch(int l, int r){
    while (l < r) {
        int mid = (long long)(l + r + 1) >> 1; //相当于 l + r 后向上取整，否则陷入死循环，例如 此时 l = 1, r = 2, 那么探针就会一直落到左区间 l 上（l + r >> 1 =  1），下一步 l = mid = l,就会死循环
        if (/* 探针落在左区间 */) l = mid;
        else r = mid - 1;
    }
    return l;
}

快速幂

int qpow(int a, int b, int mod){
    int ans = 1 % mod;
    while(b != 0) {
        if (b & 1) ans = (long long)(ans * a) % mod;
        b >>= 1;
        a = (long long)(a * a) % mod;
    }
    return ans;
}

最大公因数，最小公倍数

//最大公因数，公约数
int gcd(int a, int b){
    int m;
    do{
        m = a % b;
        a = b;
        b = m;
    }while(m != 0);
    return a;
}

//最小公倍数
int lcm(int a, int b) {
    return a / gcd(a, b) * b;
}

#include <bits/stdc++.h>

#define lb(x) (x & (-x))

using namespace std;

const int Max = 1010;
int arr[Max] = {0}, tree[Max] = {0}, n;

void build() {
    for (int i = 1; i <= n; i++) {
        tree[i] += arr[i];
        int j = i + lb(i);
        if (j <= n) tree[j] += tree[i];
    }
}

int Sum(int indx) {
    int ans = 0;

  • 区间修改与区间求和
#include <bits/stdc++.h>

#define lb(x) (x & (-x))

using namespace std;

const int Max = 1e5 + 10;
int arr[Max], tree1[Max], tree2[Max], sum[Max], n;

void update_(int indx, int v) {
    for (int i = indx; i <= n; i += lb(i)) {
        tree1[i] += v;
        tree2[i] += v * indx;
    }
}

int Sum_(int indx, int tree[]) {
    int res = 0;
    for (int i = indx; i > 0; i -= lb(i)) {
        res += tree[i];
    }
    return res;
}

void update(int l, int r, int v) {
    update_(l, v);
    update_(r + 1, -v);
}
//获取数组
int getArr(int indx) {
    return arr[indx] + Sum_(indx, tree1);
}
//获取区间的和
int getSum(int l, int r) {
    int res = sum[r] + (r + 1) * Sum_(r, tree1) - Sum_(r, tree2);
    res -= sum[l - 1] + l * Sum_(l - 1, tree1) - Sum_(l - 1, tree2);
    return res;
}

int main() {
    return 0;
}
  • 维护区间最大值
#include <bits/stdc++.h>
#define lb(x) (x & (-x))

using namespace std;

const int M = 1e6 + 10;
int arr[M], tree[M], n;

void build() {
    for (int i = 1; i <= n; i++) {
        tree[i] = arr[i];
        for (int j = 1; j < lb(i); j <<= 1) {
            tree[i] = max(tree[i], tree[i - j]);
        }
    }
}

int getMax(int l, int r) {
    int res = arr[r];
    while (r >= l) {
        res = max(arr[r], res);
        for (r--; r - lb(r) >= l; r -= lb(r)) {
            res = max(tree[r], res);
        }
    }
    return res;
}

//单点修改
void update(int indx) {
    while (indx <= n) {
        tree[indx] = arr[indx];
        for (int i = 1; i < lb(indx); i <<= 1) {
            tree[indx] = max(tree[indx], tree[indx - i]);
        }
        indx += lb(indx);
    }
}

int main() {
    return 0;

线段树

#include <bits/stdc++.h>

using namespace std;

const int M = 1e5 + 10;
int arr[M], tree[M << 2], lazy[M << 2], n;

inline void pushup(int node) {
    tree[node] = tree[node << 1] + tree[node << 1 | 1];
}

inline void pushdown(int a, int b, int node) {
    int mid = (a + b) >> 1;
    tree[node << 1] += (mid - a + 1) * lazy[node];
    tree[node << 1 | 1] += (b - mid) * lazy[node];
    lazy[node << 1] += lazy[node];
    lazy[node << 1 | 1] += lazy[node];
    lazy[node] = 0;
}

void build(int a, int b, int node) {
    if (a == b) {
        tree[node] = arr[a];
        return;
    }
    int mid = (a + b) >> 1;
    build(a, mid, node << 1);
    build(mid + 1, b, node << 1 | 1);
    pushup(node);
}

void update(int L, int R, int v, int a = 1, int b = n, int node = 1) {
    if (L <= a && b <= R) {
        tree[node] = (a - b + 1) * v;
        lazy[node] += v;
        return;
    }
    if (lazy[node])
        pushdown(a, b, node);
    int mid = (a + b) >> 1;
    if (L <= mid)
        update(L, R, v, a, mid, node << 1);
    if (R > mid)
        update(L, R, v, mid, b, node << 1 | 1);
    pushup(node);
}
int getAsk(int L, int R, int a = 1, int b = n, int node = 1) {
    if (L <= a && b <= R) {
        return tree[node];
    }
    if (lazy[node])
        pushdown(a, b, node);
    int mid = (a + b) >> 1, res = 0;
    if (L <= mid)
        res = getAsk(L, R, a, mid, node << 1);
    if (R > mid)
        res += getAsk(L, R, mid, b, node << 1 | 1);
    return res;
}

int main() {
    return 0;
}

简单的整数哈希（二进制版线性探查）

  • 一次刷题时看到一位不知名大佬的hash部分代码，试了一下，速度很快
  • 原理有点像线性探查法，哈希的方法就是对要哈希的整数与上（&）哈希最大值（H）
const int H = 0xfffff;

struct Hash{
    int real[H + 5], val[H + 5];
    void clear() { memset(val, 0, sizeof val); }
    int& operator[] (const int& num) {
        int i = num & H;
        while (val[i] && real[i] != num) i = (i + 1) & H;
        real[i] = num;
        return val[i];
    }
}h;

dijkstra(堆优化版)

  • 普通版本慢，而且优化版本代码量也不长
  • 下面是用邻接表实现的，顺便写上邻接表代码
#include <bits/stdc++.h>

using namespace std;

typedef pair<int, int> pii;

const int maxn = 1e5, maxm = 2e5;

int n, m, s; //n 为顶点数， m 为边数
int head[maxn], cnt, dis[maxn];
bool vis[maxn];
void init() {
    memset(head, -1, sizeof head);
    memset(dis, 0x3f, sizeof dis);
    memset(vis, false, sizeof vis);
    cnt = 0;
}
struct edges {
    int to, w, next;
    edges(int to = 0, int w = 0, int next = -1) : to(to), w(w), next(next) {}
} edge[maxm];

inline void add_edges(int u, int v, int w) {
    edge[++cnt] = edges(v, w, head[u]);
    head[u] = cnt;
}

void dijkstra() {
    dis[s] = 0;
    priority_queue<pii> heap;
    heap.push(make_pair(0, s));
    while (heap.size()) {
        int u = heap.top().second;
        heap.pop();
        if (vis[u]) continue;
        vis[u] = true;
        for (int i = head[u]; ~i; i = edge[i].next) {
            int v = edge[i].to, w = edge[i].w;
            if (dis[v] > dis[u] + w) { //松弛
                dis[v] = dis[u] + w;
                heap.push(make_pair(-dis[v], v));
            }
        }
    }
}

int main() {
    init();
    scanf("%d%d%d", &n, &m, &s);
    for (int i = 0; i < m; i++) {
        int u, v, w;
        scanf("%d%d%d", &u, &v, &w);
        add_edges(u, v, w);
    }
    return 0;
}

bellman - ford

  • 普通版（一般够用）
  • 注意：如果让判断无法到达，则需要再松弛时加一个判断dis[v] != INF
  • 如果没让判断负环，课在松弛循环前加上flag判断是否有松弛过，没有则直接退出bf
#include <bits/stdc++.h>

using namespace std;
const int maxn = 1e5, maxm = 2e5;
int n, m, s; // n 为顶点数， m 为边数
int dis[maxn];

struct edges {
    int u, v, w;
    edges(int u = 0, int v = 0, int w = 0) : u(u), v(v), w(w) {}
} edge[maxm];

bool bf() {
    for (int k = 1; k < n; k++) {
        for (int i = 1; i < n; i++) {
            if (dis[edge[i].v] > dis[edge[i].u] + edge[i].w) {//松弛
                dis[edge[i].v] = dis[edge[i].u] + edge[i].w;
            }
        }
    }
    for (int i = 1; i < n; i++) {
        if (dis[edge[i].v] > dis[edge[i].u] + edge[i].w) {
            return false;
        }
    }
    return true;
}

int main() {
    scanf("%d%d%d", &n, &m, &s);
    memset(dis, 0x3f, sizeof dis);
    for (int i = 0; i < m; i++) {
        int u, v, w;
        scanf("%d%d%d", &u, &v, &w);
        edge[i] = edges(u, v, w);
    }
    dis[s] = 0;
    if (bf()) {
        printf("没有负环");
    } else {
        printf("有负环");
    }
    return 0;
}
  • 队列优化版（即spfa算法）
#include <bits/stdc++.h>

using namespace std;

const int maxn = 1e5, maxm = 2e5;
int n, m, s, dis[maxn], num[maxn], head[maxn], cnt; //num 数组时判断是否有负环
bool inq[maxn];

void init() {
    memset(inq, false, sizeof inq);
    memset(dis, 0x3f, sizeof dis);
    memset(num, 0, sizeof num);
    memset(head, -1, sizeof head);
    cnt = 0;
}

struct edges {
    int to, w, next;
    edges(int to = 0, int w = 0, int next = -1) : to(to), w(w), next(next) {}
} edge[maxm];

inline void add_edges(int u, int v, int w) {
    edge[++cnt] = edges(v, w, head[u]);
    head[u] = cnt;
}

bool spfa() {
    queue<int> q;
    q.push(s);
    inq[s] = true;
    num[s]++;
    while (q.size()) {
        int u = q.front();
        q.pop();
        inq[u] = false;
        for (int i = head[u]; ~i; i = edge[i].next) {
            int v = edge[i].to, w = edge[i].next;
            if (dis[v] > dis[u] + w) {
                dis[v] = dis[u] + w;
                if (!inq[v]) {
                    q.push(v);
                    inq[v] = true;
                    num[v]++;
                    if (num[v] >= n) return false; //如果从1号点到x的最短路中包含至少n个点（不包括自己），则存在环
                }
            }
        }
    }
    return true;
}

int main() {
    scanf("%d%d%d", &n, &m, &s);
    for (int i = 0; i < m; i++) {
        int u, v, w;
        scanf("%d%d%d", &u, &v, &w);
        add_edges(u, v, w);
    }
    dis[s] = 0;
    if (spfa()) {
        printf("没有负环");
    } else {
        printf("有负环");
    }
    return 0;
}

floyd

#include <bits/stdc++.h>

using namespace std;

const int M = 2e2;
int n, m; //顶点数和边数
int G[M][M];

void floyd() {
    for (int k = 1; k <= n; k++) {
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= n; j++) {
                if (G[i][j] > G[i][k] + G[k][j]) {
                    G[i][j] = G[i][k] + G[k][j];
                }
            }
        }
    }
}

int main() {
    scanf("%d%d", &n, &m);
    for (int i = 0; i < m; i++) {
        int u, v, w;
        scanf("%d%d%d", &u, &v, &w);
        G[u][v] = w;
    }
    floyd();
    return 0;
}
